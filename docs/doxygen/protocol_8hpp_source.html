201inline std::vector<uint8_t> make_packet(uint32_t type,
202 uint32_t sender,
203 uint32_t target,
204 const void* data,
205 uint32_t data_len)
206{
207 std::vector<uint8_t> buf(HEADER_SIZE + data_len);
208 PacketHeader* h = reinterpret_cast<PacketHeader*>(buf.data());
209 h->magic = MAGIC;
210 h->msg_type = type;
211 h->sender_id = sender;
212 h->target_id = target;
213 h->data_len = data_len;
214 if (data && data_len > 0)
215 memcpy(buf.data() + HEADER_SIZE, data, data_len);
216 return buf;
217}
